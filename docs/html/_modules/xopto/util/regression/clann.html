ACTIVATION_FUNCTIONS = ('relu', 'tanh', 'exponential', 'softmax',
                        'softplus', 'sigmoid', 'softsign', 'selu')
""" Tuple of implemented activation functions. """


CL_ACTIVATION_FUNCTIONS = \
'''
inline void activation_relu(
    mc_fp_t max_value, mc_fp_t negative_slope, mc_fp_t threshold,
    mc_fp_t *in_out, mc_size_t n)
{
    for(mc_size_t pos=0; pos < n; ++pos){
        in_out[pos] = (in_out[pos] > max_value) ? max_value : in_out[pos];
        in_out[pos] = (in_out[pos] >= threshold) ?
            in_out[pos] : negative_slope*(in_out[pos] - threshold);
    }
};

inline void activation_tanh(
    mc_fp_t *in_out, mc_size_t n)
{
    for(mc_size_t pos = 0; pos < n; ++pos){
        in_out[pos] = mc_tanh(in_out[pos]);
    };
};

inline void activation_elu(
    mc_fp_t alpha, mc_fp_t *in_out, mc_size_t n)
{
    for(mc_size_t pos=0; pos < n; ++pos){
        in_out[pos] = (in_out[pos] >= FP_0) ?
            in_out[pos] : alpha*(mc_exp(in_out[pos]) - FP_1);
    }
};

inline void activation_exponential(
    mc_fp_t *in_out, mc_size_t n)
{
    for(mc_size_t pos=0; pos < n; ++pos){
        in_out[pos] = mc_exp(in_out[pos]);
    }
};

inline void activation_softmax(
    mc_fp_t *in_out, mc_size_t n)
{
    mc_fp_t total = FP_0;
    for(mc_size_t pos = 0; pos < n; ++pos){
        in_out[pos] = mc_exp(in_out[pos]);
        total += in_out[pos];
    };

    mc_fp_t inv_total = mc_fdiv(FP_1, total);
    for(mc_size_t pos = 0; pos < n; ++pos){
        in_out[pos] *= inv_total;
    };
};

inline void activation_softplus(
    mc_fp_t *in_out, mc_size_t n)
{
    for(mc_size_t pos = 0; pos < n; ++pos)
        in_out[pos] = mc_fdiv(FP_1, (FP_1 + mc_exp(-in_out[pos])));
};

inline void activation_sigmoid(
    mc_fp_t *in_out, mc_size_t n)
{
    for(mc_size_t pos = 0; pos < n; ++pos)
        in_out[pos] = mc_log(mc_exp(in_out[pos]) + FP_1);
};

inline void activation_softsign(
    mc_fp_t *in_out, mc_size_t n)
{
    for(mc_size_t pos = 0; pos < n; ++pos)
        in_out[pos] = mc_fdiv(in_out[pos], mc_fabs(in_out[pos] + FP_1));
};

inline void activation_selu(
    mc_fp_t *in_out, mc_size_t n)
{
    mc_fp_t const scale = FP_LITERAL(1.05070098);
    mc_fp_t const alpha = FP_LITERAL(1.67326324);
 
    for(mc_size_t pos = 0; pos < n; ++pos)
        in_out[pos] = (in_out[pos] >= FP_0) ?
            scale*in_out[pos] : scale*alpha*(mc_exp(in_out[pos]) - FP_1);
};
'''
""" OpenCL source code of the implemented activation functions. """


CL_COMMON = \
'''
void inline fp_copy(
    mc_fp_t *dest, mc_fp_t const *src, mc_size_t n)
{
    for(mc_size_t pos=0; pos < n; ++pos)
        dest[pos] = src[pos];
};

void inline fp_copy_from_global(
    mc_fp_t *dest, __global mc_fp_t const *src, mc_size_t n)
{
    for(mc_size_t pos=0; pos < n; ++pos)
        dest[pos] = src[pos];
};

void inline fp_copy_to_global(
    __global mc_fp_t *dest, mc_fp_t const *src, mc_size_t n)
{
    for(mc_size_t pos=0; pos < n; ++pos)
        dest[pos] = src[pos];
};
'''
""" OpenCL source code of support functions. """


CL_KERNEL_TEMPLATE = \
'''
__kernel void {prefix}model_kernel(
        __global mc_fp_t const *input,
        __global mc_fp_t *output,
        mc_size_t num_samples)
{{
    if (get_global_id(0) < num_samples){{
        mc_fp_t one_input[{num_inputs}];
        mc_fp_t one_output[{num_outputs}];

        fp_copy_from_global(one_input, input + {num_inputs}*get_global_id(0), {num_inputs});
        {prefix}input_pp_apply(one_input);
        {prefix}model(one_input, one_output);
        {prefix}output_pp_undo(one_output);
        fp_copy_to_global(output + {num_outputs}*get_global_id(0), one_output, {num_outputs});
    }}
}};
'''
"""
OpenCL kernel function template as a python format string with keyword
parameters.

Parameters
----------
    prefix: str
        Prefix for the OpenCL kernel and the related support functions that
        implement the model.
    num_inputs: int
        The number of input variables.
    num_outputs: int
        The number of output variables
"""

    '''
    Returns the input value if it is not None, else returns the default value.

    Parameters
    ----------
    value: float
        Input floating point number or None.
    default: float
        Default value that is returned if the input value is None.

    Returns
    -------
    fp: float
        Input value if it is not None, else the default value.

[docs]    def render(self, ctx: ClWorker, models: List[Tuple[ann.Model, str]]) \
            -> List[str]:
        '''
        Render/generate OpenCL source code for the provided dense models.

        Parameters
        ----------
        ctx: ClWorker
            OpenCL worker that will be used to run the OpenCL-based
            implementation of the model.
        models: List[Tuple[ann.Model, str]]
            A list of ANN models and corresponding prefixes that will be used
            to produce OpenCL code/kernels.

        Returns
        -------
        code: List[str]
            Source code lines. 
        '''
        prefixes = [prefix for _, prefix in models]
        if len(set(prefixes)) != len(prefixes):
            raise ValueError('Prefixes of ANN models must be unique!')
            
        pp_code = {}
        lines = [self._cl_base(), '',
                 '/* Common code */',
                 CL_COMMON, '',
                 '/* Activation functions */',
                 CL_ACTIVATION_FUNCTIONS, '']

        # All implementations of preprocessors go to a common place
        lines.extend(['/* Preprocessors */', ''])
        for model, prefix in models:
            self._cl_pp_implementation(ctx, model.topology.pp_input, pp_code)
            self._cl_pp_implementation(ctx, model.topology.pp_input, pp_code)
        lines.extend(pp_code.values())
        lines.append('')

        # Generate code for the individual models
        for model, prefix in models:
            lines.extend(['/* Preprocessor calls */', ''])
            lines.extend(
                self._cl_render_pp(ctx, model, prefix, self._indent, self._width)
            )

            lines.extend(['/* Dense network */', ''])
            lines.extend(
                self._cl_render_dense(ctx, model, prefix, self._indent, self._width)
            )

            lines.extend(['/* Kernel */'])
            num_inputs = model.layers[0].get_weights()[0].shape[0]
            num_outputs = model.layers[-1].get_weights()[0].shape[1]
            lines.append(
                CL_KERNEL_TEMPLATE.format(prefix=prefix,
                                        num_inputs=num_inputs,
                                        num_outputs=num_outputs)
            )

        return lines


    def _cl_base(self) -> str:
        '''
        Prepare and return the base code of the OpenCL implementation that
        defines the basic data types, math functions and handles the source
        code options.

        Returns
        -------
        src: str
            The OpenCL source code.
        '''
        code = ''
        f = os.path.join(UTIL_PATH, 'regression', 'kernel', 'clann.template.h')
        with open(f, 'r') as fid:
            code = fid.read()

        return code

    def _cl_pp_implementation(self, ctx: ClWorker,
                              preprocessor: pp.Preprocessor,
                              code: Dict[pp.PreprocessorFunction, str] = None) \
            -> Dict[pp.PreprocessorFunction, str]:
        '''
        Generates OpenCL source code that implements the preprocessor functions
        of the ANN model.

        Parameters
        ----------
        ctx: ClWorker
            OpenCL worker that will be used to run the OpenCL-based
            implementation of the model.
        preprocessor: pp.Preprocessor
            Instance of a preprocessor used by the model.
        code: Dict[pp.PreprocessorFunction, str]
            Any existing source code that implements preprocessor functions.

        Returns
        -------
        code: Dict[pp.PreprocessorFunction, str]
            Source code of the individual preprocessor functions in a dict.
        '''
        if code is None:
            code = {}
        for pp_item in preprocessor.items:
            for func in pp_item.sequence:
                T = type(func)
                if T not in code:
                    func_code = func.fetch_cl_implementation(ctx)
                    if func_code is not None:
                        code[T] = func_code

        return code

    def _cl_render_pp(self, ctx: ClWorker, model: ann.Model, prefix: str = None,
                     indent: str = None, width: int = 80) -> List[str]:
        '''
        Render OpenCL code for a preprocessor call (NOT preprocessor
        implementation).

        Parameters
        ----------
        ctx: ClWorker
            OpenCL worker that will be used to run the OpenCL-based
            implementation of the model.
        model: ann.Model
            The ANN model that will be converted to OpenCL.
        prefix: str
            Prefix for the OpenCL kernel and related support functions that
            implement the model.
        indent: str
            Indentation of the source code.
        width: int
            Maximum line width of the source code.

        Returns
        -------
        code: List[str]
            Source code lines that implement the preprocessor call.
        '''
        prefix = prepare_prefix(prefix)

        if indent is None:
            indent = self.DEFAULT_INDENT

        layers = model.layers
        num_inputs = layers[0].get_weights()[0].shape[0]
        num_outputs = layers[-1].get_weights()[0].shape[1]

        lines = []
        topology = model.topology
        for dir, render in (('input', topology.pp_input.cl_render_call),
                            ('output', topology.pp_output.cl_render_call)):
            in_out_prefix = prefix + dir + '_'
            for undo in (False, True):
                lines.extend(
                    render(ctx, in_out_prefix, 'data', num_inputs,
                           indent=indent, width=width, undo=undo)
                )
                lines.append('')

        return lines

    def _cl_render_dense(self, ctx: ClWorker, model: ann.Model,
                         prefix: str = None,
                         indent: str = None, width: int = 80) -> List[str]:
        '''
        Render OpenCL code for the dense ANN network.

        Parameters
        ----------
        ctx: ClWorker
            OpenCL worker that will be used to run the OpenCL-based
            implementation of the model.
        model: ann.Model
            The ANN model that will be converted to OpenCL.
        prefix: str
            Prefix for the OpenCL kernel and related support functions that
            implement the model.
        indent: str
            Indentation of the source code.
        width: int
            Maximum line width of the source code.

        Returns
        -------
        code: List[str]
            Source code lines that implement the dense ANN network.
        '''
        prefix = prepare_prefix(prefix)

        if indent is None:
            indent = self.DEFAULT_INDENT

        input_var = 'input'
        output_var = 'output'
        layers = model.layers
        num_inputs, num_outputs = layers[0].get_weights()[0].shape[:2]
        num_layers = len(layers)
        lines = []
        lines.extend([
            'void {prefix}model('.format(prefix=prefix),
            '{indent}mc_fp_t const *input, mc_fp_t *output)'.format(
                indent=indent),
            '{'
        ])

        body = []
        for layer_index, layer in enumerate(layers):
            allocate_output_var = False

            if layer_index == 0:
                layer_input_var = 'input'
            else:
                layer_input_var = 'output_{}'.format(layer_index - 1)

            alloc_output_var = False
            if layer_index == num_layers - 1:
                layer_output_var = 'output'
            else:
                allocate_output_var = True
                layer_output_var = 'output_{}'.format(layer_index)

            body.append('/* Layer {} of {} */'.format(
                layer_index + 1, num_layers)
            )
            if allocate_output_var:
                body.append(
                    'mc_fp_t {layer_output_var}[{num_outputs}];'.format(
                        layer_output_var=layer_output_var,
                        num_outputs = layer.get_weights()[0].shape[1]
                    )
                )

            body.extend(
                self._cl_render_layer(ctx, layer,
                                      layer_input_var,
                                      layer_output_var,
                                      width=width - len(indent), indent=indent)
            )
            body.append('')

        for line in body:
            lines.append(indent + line)

        lines.extend(['};', ''])

        return lines

    def _cl_render_layer(self, ctx: ClWorker, layer: object, 
                         input_var: str, output_var: str,
                         indent: str, width: int) -> List[str]:
        '''
        Render OpenCL code for the dense ANN network.

        Parameters
        ----------
        ctx: ClWorker
            OpenCL worker that will be used to run the OpenCL-based
            implementation of the model.
        layer: object
            Native representation of the model layer.
        input_var: str
            Name of the variable for the input data vector.
        output_var: str
            Name of the variable for the output data vector.
        indent: str
            Indentation of the source code.
        width: int
            Maximum line width of the source code.

        Returns
        -------
        code: List[str]
            Source code lines that implement the network layer.
        '''
        all_weights = layer.get_weights()
        n_in, n_out = all_weights[0].shape
        num_weights = n_in
        lines = []
        T = ctx.types
        for output_index in range(n_out):
            weights = all_weights[0][:, output_index].flat
            lines.append(
                '{output_var}[{index}] ='.format(output_var=output_var,
                                                 index=output_index)
            )
            line = str(indent)
            for weight_index, weight in enumerate(weights):
                item = '{input_var}[{index}]*{weight}'.format(
                    input_var=input_var, index=weight_index,
                    weight=T.fp_str(weight)
                )
                if weight_index == num_weights - 1:
                    if len(line) + len(item) + 1 < width:
                        line += item + ';'
                    else:
                        lines.append(line)
                        line = '{indent}{item};'.format(
                            indent=indent, item=item)
                else:
                    if len(line) + len(item) + 3 < width:
                        line += '{} + '.format(item)
                    else:
                        lines.append(line)
                        line = '{indent}{item} + '.format(
                            indent=indent, item=item)
            lines.append(line)

            if len(all_weights) > 1:
                bias = all_weights[1]
                line = '{output_var}[{index}] += {bias};'.format(
                    output_var=output_var, index=output_index,
                    bias=T.fp_str(bias[output_index]))
                lines.append(line)

        lines.extend(
            self._cl_render_activation(ctx, layer, output_var,
                                       indent=indent, width=width)
        )

        return lines

    def _cl_render_activation(self, ctx: ClWorker, layer, inout_var:str,
                              indent: str, width: int) -> List[str]:
        '''
        Render activation function of the layer.

        Parameters
        ----------
        ctx: ClWorker
            OpenCL worker that will be used to run the OpenCL-based
            implementation of the model.
        layer: object
            Native representation of the model layer.
        inout_var: str
            Name of the variable for the input/output data vector.
        indent: str
            Indentation of the source code.
        width: int
            Maximum line width of the source code.

        Returns
        -------
        code: List[str]
            Source code lines that implement the activation function
            of the network layer.
        '''
        n = layer.get_weights()[0].shape[1]
        activation = layer.activation
        activation_name = layer.activation.__name__
        if activation_name not in ACTIVATION_FUNCTIONS:
            raise ValueError('Unexpected activation function "{}"!'.format(
                activation_name))

        lines = []
        function = 'activation_{:s}'.format(activation_name)
        T = ctx.types
        lines.append(
            '{function}('.format(function=function)
        )
        if activation_name == 'relu':
            lines.append(
                '{indent}{max_value:s}, {negative_slope:s}, '
                '{threshold:s},'.format(
                    indent=indent,
                    max_value=T.fp_str(
                        get_fp(getattr(activation, 'max_value', None), np.inf)
                    ),
                    negative_slope=T.fp_str(
                        get_fp(getattr(activation, 'negative_slope', None), 0.0)
                    ),
                    threshold=T.fp_str(
                        get_fp(getattr(activation, 'threshold', None), 0.0)
                    )
                )
            )
        elif activation == 'elu':
            lines.append(
                '{indent}{alpha:s},'.format(
                    indent=indent,
                    alpha=T.fp_str(
                        get_fp(getattr(activation, 'alpha', None), 1.0)
                    ),
                )
            )
        lines.append(
            '{indent}{inout_var}, {n}'.format(
                indent=indent, inout_var=inout_var, n=n)
        )
        lines.append(');')

        return lines

    def __init__(self, model: str or ann.Model, prefix: str = '',
                 options: List[mcoptions.McOption] = None,
                 renderer: 'Renderer' = None,
                 *args, **kwargs):
        '''
        OpenCL implementation of a dense model.

        model: ann.Model or str
            Dense model instance. If a str instance, load the model
            from the specified file.
        prefix: str
            Prefix for the OpenCL kernel and related support functions that
            implement the model.
        options: List[mcoptions.McOption]
            List of options applied to the source code.
        renderer: Renderer
            An instance of rendering engine that will be used to generate
            OpenCL source code representing the model.
        args: tuple
            Positional arguments passed to the base class
            constructor :py:class:`xopto.mcbase.mcworker.ClWorker` class.
        kwargs: dict
            Keyword arguments passed to the base class
            constructor :py:class:`xopto.mcbase.mcworker.ClWorker` class.
        '''
        super().__init__(*args, **kwargs)
        self._cl_src = self._cl_exec = None

        if isinstance(model, str):
            model = ann.Model.load(model)

        if options is None:
            options = []

        if renderer is None:
            renderer = Renderer()

        prefix = prepare_prefix(prefix)

        self._model = model
        self._renderer = renderer
        self._kernel_options = options
        self._prefix = prefix

        self._kernel_names = []

        self.ninput = model.ninput
        self.noutput = model.noutput
        self.layers = model.layers
        self.topology = model.topology

        # the latest kernel timing report and the number of used threads
        self._run_report = {'upload': None, 'download': None,
                            'execution': None, 'items': None}


[docs]    def build(self, exportsrc: str = None, verbose: bool = False) -> float:
        '''
        Force build the OpenCL source and executable. The source and executable
        are otherwise build on the first use.

        Parameters
        ----------
        exportsrc: str
            Export the generated OpenCL source code to this file before
            starting the build process.
        verbose: bool
            Enables verbose reporting.
        '''
        if self._cl_exec is None:
            self._build_exec(exportsrc=exportsrc, verbose=verbose)


    def _build_src(self, verbose: bool = False):
        '''
        Build the OpenCL source code.

        Parameters
        ----------
        verbose: bool
            Enables verbose reporting.
        '''
        kernel_options = []
        for item in self._kernel_options:
            if isinstance(item, mcoptions.McOption):
                kernel_options.extend(item.fetch_cl_options(self))
            elif isinstance(item, (tuple, list)):
                kernel_options.append(tuple(item))
            else:
                raise ValueError(
                    'Kernel options must be a list of McOption or a list '
                    'of tuples!')
        resolved_options = mcoptions.resolve_cl_options(
            self._types().fetch_cl_options(self),
            kernel_options
        )

        context = {}
        context['mc'] = {'options': mcoptions.make_defines(resolved_options)}

        template = jinja2.Template(
            '\n'.join(
                self._renderer.render(self, [(self._model, self._prefix)])
            )
        )

        # save all the kernel options for later use
        self._cl_src_options = dict(resolved_options)

        # Render the OpenCL source file.
        self._cl_src = template.render(context)

    def _build_exec(self, exportsrc: bool or str = None, verbose: bool = False):
        '''
        Build the OpenCL executable from the source code.

        Parameters
        ----------
        exportsrc: bool or str
            Optional filename that will be used to save the source code.
        verbose: bool
            Enables verbose reporting.

        Returns
        -------
        buildtime: float
            Time required to build the OpenCL executable.
        '''
        # Rebuild from source if required - first call of the run method.
        tb = time.perf_counter()
        # Generate / render the source code for this simulator instance.
        if self._cl_src is None:
            self._build_src(verbose=verbose)
        if exportsrc is not None and exportsrc:
            filename = None
            if isinstance(exportsrc, str):
                filename = exportsrc
            self.export_src(filename)
        # Compile the OpenCL source code.
        self._cl_exec = self.cl_build(self._cl_src, verbose)
        for item in self._cl_exec.all_kernels():
            self._kernel_names.append(item.function_name)

        # options=['-cl-opt-disable', '-Werror']
        buildtime = time.perf_counter() - tb

        if verbose:
            print('OpenCL engine created in {:.3f} ms.'.format(buildtime*1e3))

        return buildtime

            (samples, noutput).
        '''
        return np.empty((samples, self._model.noutput),
                        dtype=self.types.np_float)



[docs]    def predict(self, input: np.ndarray, output: np.ndarray = None,
                wgsize: int = None, verbose: bool = False) \
            -> np.ndarray:
        '''
        Compute predictions for the input data array using OpenCL
        implementation of the model. The input data array is first transferred
        to the OpenCL device. After running the OpenCL model kernel, the
        prediction results are downloaded into the output data array.

        Parameters
        ----------
        input: np.ndarray
            Input data as a 2D numpy array with shape (samples, n), where n is
            the number of inputs and samples the number of samples in the
            input array.
        output: np.ndarray
            Optional data array for the predictions that must have sufficient
            space and be of the same floating point type as used
            in the OpenCL kernel.
        wgsize: int
            Local work group size. If None (default), the optimal work group
            size is determined automatically.
        verbose: bool
            Enables verbose reporting.

        Returns
        -------
        output: np.ndarray
            Data array with predictions.
        '''
        t_start = time.perf_counter()

        if self._cl_exec is None:
            self._build_exec(verbose=verbose)

        input = np.asarray(input, dtype=self.types.np_float)

        if input.ndim != 2:
            raise ValueError('The input data must be a 2D array!')

        num_inputs = self._model.ninput
        if input.shape[1] != num_inputs:
            raise ValueError('The number of inputs in the data array ({}) does '
                             'not match the number of inputs ({}) '
                             'in the model!'.format(
                                 input.shape[1], num_inputs))

        num_samples = input.shape[0]

        if output is not None and output.dtype != self.types.np_float:
            raise TypeError('The output array must have the same floating '
                            'point type as the OpenCL kernel!')

        num_outputs = self._model.noutput

        if output is None:
            output = np.empty((num_samples, num_outputs),
                              dtype=self.types.np_float)

        if output.size != num_samples*num_outputs:
            raise ValueError('The output array must be of shape '
                             '(nsamples, noutput)!')

        # configure the local and global work group size
        if wgsize is None:
            local_wg = None
            global_wg = [num_samples]
        else:
            wgsize = int(wgsize)
            local_wg = [wgsize]
            global_wg = [((num_samples + wgsize - 1)//wgsize)*wgsize]

        cl_buffers = self.cl_buffers
        cl_queue = self.cl_queue

        self.cl_r_buffer('input', input)
        self.cl_w_buffer('output',
                         size=num_samples*num_outputs*output.itemsize)
        t_upload = time.perf_counter()

        kernel_name = self._kernel_names[0]
        getattr(self.cl_exec, kernel_name)(
            cl_queue, global_wg, local_wg,
            cl_buffers['input'],
            cl_buffers['output'],
            np.dtype(self._types.np_size).type(num_samples),
        ).wait()
        t_exec = time.perf_counter()

        cl.enqueue_copy(cl_queue, output, cl_buffers['output']).wait()
        t_download = time.perf_counter()

        uploadtime = t_upload - t_start
        exectime = t_exec - t_upload
        downloadtime = t_download - t_exec

        self._run_report.update(upload=uploadtime, download=downloadtime,
                                execution=exectime, items=num_samples)

        if verbose:
            print('OpenCL model processed {:,d} items:\n'
                  '    uploaded/built in: {:10.3f} ms\n'
                  '    executed in      : {:10.3f} ms\n'
                  '    downloaded in    : {:10.3f} ms'.format(
                      num_samples, uploadtime*1e3, exectime*1e3,
                      downloadtime*1e3)
            )

        return output


    def _get_run_report(self) -> dict:
        return self._run_report

                          'Timing and performance data of the latest kernel '

                          'The number of processed items with key "items" '

                          'All values are initialized to None.')


[docs]    def export_src(self, filename: str = None) -> str:
        '''
        Export the OpenCL source to a file.

        Parameters
        ----------
        filename: str
            The OpenCL source code will be saved to this file. If a file
            name is not provided, the OpenCL source will be saved to
            :py:attr:`ClModel.AUTOGEN_PATH` directory with a name
            "<prefix>_clmodel_<random>.cl".

        Returns
        -------
        filename: str
            The filename that was used to save the OpenCL source code.
        '''
        if self._cl_src is not None:
            if filename is None:
                suffix = str(time.perf_counter()).replace('.', '')
                filename = os.path.join(
                    ClModel.AUTOGEN_PATH, '{}clmodel_{}.cl'.format(
                        self._prefix, suffix)
                )
            with open(filename, 'wt') as fp:
                fp.write(self._cl_src)

        return filename
